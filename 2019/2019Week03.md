### 使用 protocol 解耦
```swift
class People {
  var name: String!
  var age: String?
  func fastRun(speed: Int) {
    print("run at \(speed) m/s")
  }
}

protocol Run {
  func fastRun(speed: Int)
}

class Student: People, Run {
  
}

class Router {
  var runner: Run
  
  init(runner: Run) {
    self.runner = runner
  }
  
  func run() {
    self.runner.fastRun(speed: 5)
  }
}

var student = Student()
var router = Router(runner: student)
router.run()
```
