## Algorithm：
[Leetcode - Hard] 25. Reverse Nodes in k-Group
```swift
func reverseKGroup(_ head: ListNode?, _ k: Int) -> ListNode? {
    if head?.next == nil {
        return head
    }
    let dummy: ListNode? = ListNode(0)
    dummy!.next = head
    var pre = dummy, start = dummy, end = dummy
    while end != nil {
        end = pre
        for _ in 0..<k {
            end = end?.next
            if end == nil {
                return dummy?.next
            }
        }
        for _ in 1..<k {
            start = pre!.next
            pre!.next = start!.next
            start!.next = end!.next
            end!.next = start
        }
        for _ in 0..<k {
            pre = pre!.next
        }
    }
    return dummy!.next
}
```
